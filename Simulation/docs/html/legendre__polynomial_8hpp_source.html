12 struct legendre_polynomial {

14 
15  T operator()(const T &x) const {

17  t[0] = legendre_polynomial<T, 0>()(x);

19 
20  int idx;
21  for (int i = 2; i <= n; ++i) {
22  idx = i % 2;
