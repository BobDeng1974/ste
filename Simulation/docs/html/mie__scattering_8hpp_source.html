10 /*
11 * The Cornette-Shanks Mie scattering phase function, an updated Henyey-Greenstein phase function.
12 *
13 * @param cosine Cosine of input angle.
14 * @param g Scattering variation parameter.
15 * Ranges from backscattering (<0) through isotropic scattering (0) to forward scattering (>0). (-1,+1)
16 */
17 template <typename T>
18 T cornette_shanks_phase_function(const T &cosine, const T &g) {
19  auto g2 = g*g;
20  auto denom = static_cast<T>(8) * glm::pi<T>() * (static_cast<T>(2) + g2) *
21  glm::pow(static_cast<T>(1) + g2 - static_cast<T>(2) * g * cosine,
22  static_cast<T>(3) / static_cast<T>(2));
23 
24  return static_cast<T>(3) * (static_cast<T>(1) + cosine*cosine) * (static_cast<T>(1) - g2) / denom;
25 }
26 /*
27 * The Cornette-Shanks Mie scattering phase function, an updated Henyey-Greenstein phase function.
28 *
29 * @param u Normalized direction vector
30 * @param v Normalized direction vector
31 * @param g Scattering variation parameter.
32 * Ranges from backscattering (<0) through isotropic scattering (0) to forward scattering (>0). (-1,+1)
33 */
34 template <typename T>
35 T cornette_shanks_phase_function(const glm::tvec3<T> &u, const glm::tvec3<T> &v, const T &g) {
36  return cornette_shanks_phase_function(glm::dot(u, v), g);
37 }
