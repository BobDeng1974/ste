50  void spawn_worker(int schedule_on_cpu = -1) {
51  workers.emplace_back([this]() {
52  std::unique_ptr<unique_thread_pool_type_erased_task> task;
53 
54  for (;;) {
55  if (interruptible_thread::is_interruption_flag_set()) return;
56 
57  {
58  std::unique_lock<std::mutex> l(this->m);
59 
60  ++threads_sleeping;
61  this->notifier.wait(l, [&]() {
62  return interruptible_thread::is_interruption_flag_set() ||
63  (task = task_queue.pop()) != nullptr;
64  });
65  --threads_sleeping;
66  }
67 
68  active_workers.fetch_add(1, std::memory_order_relaxed);
69 
70  while (task != nullptr) {

72  if (interruptible_thread::is_interruption_flag_set())
73  return;

78  }
79  });

160  void load_balance() {
161  assert(is_main_thread());
162 
163  auto now = std::chrono::high_resolution_clock::now();
164  std::chrono::duration<float> time_since_last_pool_balance = now - last_pool_balance;
165  if (time_since_last_pool_balance.count() < .05f)
166  return;
167  last_pool_balance = now;
168 
169  float idle_frac = .0f;
170  float kernel_frac = .0f;
171  float user_frac = .0f;
172  if (!sys_times.get_times_since_last_call(idle_frac, kernel_frac, user_frac))
173  return;
174 
175  // Cleanup despawn queue
176  for (auto it = despawned_workers.begin(); it != despawned_workers.end(); ) {
177  if (it->get_future().wait_for(std::chrono::seconds(0)) == std::future_status::ready) {
178  it->get_future().get(); // Might throw
179  it = despawned_workers.erase(it);
180  }
181  else
182  ++it;
183  }
184 
185  // Check for workers with exceptions

187  if (it->get_future().wait_for(std::chrono::seconds(0)) == std::future_status::ready) {
188  it->get_future().get();

190  }
191  else
192  ++it;
193  }
194
