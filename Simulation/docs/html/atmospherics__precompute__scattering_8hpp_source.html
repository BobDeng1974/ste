72  static T height_to_lut_idx(const T &h, const T &h_max) {

107  void load(const boost::filesystem::path &path) {
108  std::ifstream ifs;
109  ifs.exceptions(ifs.exceptions() | std::ios::failbit);
110  ifs.open(path.string(), std::ios::binary | std::ios::in);
111 
112  if (!ifs)
113  throw atmospherics_lut_error("Failed opening LUT");
114 
115  if (!ifs.read(reinterpret_cast<char*>(this), sizeof(*this) / sizeof(char)))
116  throw atmospherics_lut_error("Can not read LUT");
117  ifs.close();
118 
119  if (this->sizeof_scalar != sizeof(T))
120  throw atmospherics_lut_error("Invalid scalar size");
121 
122  if (this->version != expected_version)
123  throw atmospherics_lut_error("Unsupported version");
124 
125  if (this->optical_length_dims != optical_length_size ||
126  this->scatter_dims_0 != scatter_size0 ||
127  this->scatter_dims_1 != scatter_size1 ||
128  this->scatter_dims_2 != scatter_size2)
129  throw atmospherics_lut_error("LUT size zero");
130 
131  boost::crc_32_type crc_computer;
132  crc_computer.process_bytes(reinterpret_cast<const std::uint8_t*>(&data), sizeof(data));
133  if (this->hash != crc_computer.checksum())
134  throw atmospherics_lut_error("Hash mismatch");
135  }

152  gli::texture2d lut_texture = gli::texture2d(gli::format::FORMAT_R32_SFLOAT_PACK32,
153  glm::ivec2{ lut_data_t::optical_length_size });
154 
155  auto *lut = reinterpret_cast<float*>(lut_texture.data());
156 
157  for (int y = 0; y < lut_data_t::optical_length_size; ++y) {
158  for (int x = 0; x < lut_data_t::optical_length_size; ++x) {
159  *lut = (*data->optical_length_lut(lut_idx))[x][y];
160  ++lut;
161  }
162  }

165  }
