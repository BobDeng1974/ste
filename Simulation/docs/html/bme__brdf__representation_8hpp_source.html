81  task<exitant_db_descriptor> load_bme_brdf_task(const boost::filesystem::path &bme_data) const {
82  return [=](optional<task_scheduler*> sched) {
83  exitant_db db;
84  float in_theta = -1;
85  int in_phi = -1;
86 
87  std::ifstream fs(bme_data.string(), std::ios::in);
88  if (!fs.good()) {
89  using namespace Text::Attributes;
90  ste_log_error() << Text::AttributedString("Error while reading BME database \"") + i(bme_data.string()) + "\": " + std::strerror(errno) << std::endl;
91  assert(false);
92 
93  return exitant_db_descriptor();
94  }
95 
96  std::string l;
97  while (std::getline(fs, l)) {
98  if (l[0] == '#') {
99  if (l.find("intheta") == 1) in_theta = std::stof(l.substr(1 + sizeof("intheta")), nullptr);
100  if (l.find("inphi") == 1) in_phi = std::stol(l.substr(1 + sizeof("inphi")), nullptr);
101  continue;
102  }
103 
104  assert(in_theta >= 0 && in_phi >= 0);
105 
106  std::stringstream ss(l);
107  bme_brdf_descriptor_entry entry;
108  ss >> entry;
109 
110  if (entry.brdf > .0f) {
111  entry.phi -= in_phi;
112  while (entry.phi > pBRDF::phi_max) entry.phi -= 360.f;
113  while (entry.phi < pBRDF::phi_min) entry.phi += 360.f;
114  append_entry(std::move(entry), db);
115  }
116  }
117 
118  fs.close();
119 
120  exitant_db_descriptor desc;
121  desc.db = std::move(db);
122  desc.in_theta = glm::round(in_theta);
123  return desc;
124  };
125  }

148  static void create_layer(int i, const database_type::iterator &it, common_brdf_representation &brdfdata, const glm::ivec3 &dims) {
149  float *data = reinterpret_cast<float*>(brdfdata.get_data()->data()) + dims.x * dims.y * i;
150  exitant_db &db = it->second;
151 
152  for (int j = 0; j < dims.y; ++j) {
153  float theta = glm::mix<float>(pBRDF::theta_min, pBRDF::theta_max, static_cast<float>(j) / static_cast<float>(dims.y - 1));
154 
155  for (int k = 0; k < dims.x; ++k) {
156  float phi = glm::mix<float>(pBRDF::phi_min, pBRDF::phi_max, static_cast<float>(k) / static_cast<float>(dims.x - 1));
157 
158  glm::vec3 w = BxDF<float>::omega(glm::radians(theta), glm::radians(phi));
159 
160  std::vector<bme_brdf_descriptor_entry*> entries;
161  int theta_bucket = static_cast<int>(theta / bucket_size);
162  int phi_bucket = static_cast<int>(phi / bucket_size);
163 
164  constexpr int samples = 15;
165  constexpr auto p_range = pBRDF::phi_max - pBRDF::phi_min + 1;
166  int bias = 1;
167  while (entries.size() < samples) {
168  for (int x = -bias; x <= bias; ++x) {
169  int tm = glm::clamp(theta_bucket + x, pBRDF::theta_min, pBRDF::theta_max);
170  int pm = phi_bucket + bias <= pBRDF::phi_max ? phi_bucket + bias : phi_bucket + bias - p_range;
171  int pn = phi_bucket - bias >= pBRDF::phi_min ? phi_bucket - bias : phi_bucket - bias + p_range;
172  for (auto &e : db[tm][pm])
173  entries.push_back(&e);
174  if (bias)
175  for (auto &e : db[tm][pn])
176  entries.push_back(&e);
177  }
178  for (int x = -bias + 1; x < bias; ++x) {
179  int tm = glm::clamp(theta_bucket + bias, pBRDF::theta_min, pBRDF::theta_max);
180  int tn = glm::clamp(theta_bucket - bias, pBRDF::theta_min, pBRDF::theta_max);
181  int pm = phi_bucket + x;
182  if (pm > pBRDF::phi_max) pm -= p_range;
183  if (pm < pBRDF::phi_min) pm += p_range;
184  for (auto &e : db[tm][pm])
185  entries.push_back(&e);
186  if (bias)
187  for (auto &e : db[tn][pm])
188  entries.push_back(&e);
189  }
190 
191  ++bias;
192  }
193  for (auto &e : db[theta_bucket][phi_bucket])
194  entries.push_back(&e);
195 
196  assert(entries.size());
197 
198  using closest_vector_type = std::vector<std::pair<float, bme_brdf_descriptor_entry*>>;
199  closest_vector_type closest;
200  float max_w = .0f, min_w = 1.f;
201  for (auto &entry : entries) {
202  float d = glm::dot(entry->v, w);
203  max_w = glm::max(d, max_w);
204  min_w = glm::min(d, min_w);
205 
206  auto pair = std::make_pair(d, entry);
207  if (closest.size() > samples && closest[closest.size() - 1].first > d) continue;
208  closest.insert(std::lower_bound(closest.begin(), closest.end(), pair, [](const closest_vector_type::value_type &lhs, const closest_vector_type::value_type &rhs) {
209  return lhs.first > rhs.first;
210  }), pair);
211 
212  if (closest.size() > samples) closest.pop_back();
213  }
214 
215  float tweight = .0f;
216  float brdf = .0f;
217  for (auto &e : closest) {
218  float weight = max_w - min_w > 0 ? glm::pow((e.first - min_w) / (max_w - min_w), 2.f) : 1.f;
219  brdf += e.second->brdf * weight;
220  tweight += weight;
221  }
222  brdf /= tweight;
223 
224  data[j * dims.x + k] = closest[0].second->brdf;
225  }
226  }
227  }

271 
272  for (auto &f : futures)
273  f.wait();
274
