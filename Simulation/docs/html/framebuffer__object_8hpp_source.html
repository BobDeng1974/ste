222  void set_color_output_binding_index(unsigned attachment_index, const color_layout_binding &binding) {
223  unsigned color_output = binding == layout_binding_none<fbo_color_attachment_layout_binding_type>() ?
224  GL_NONE :
225  GL_COLOR_ATTACHMENT0 + binding.binding_index();
226 
227  if (draw_buffers.size() < attachment_index + 1) draw_buffers.resize(attachment_index + 1, GL_NONE);
228  bool changed = draw_buffers[attachment_index] != color_output;
229  draw_buffers[attachment_index] = color_output;
230 
231  if (changed) {
232  for (unsigned i = 0; i < draw_buffers.size(); ++i) {
233  if (i == attachment_index) continue;
234  if (draw_buffers[i] == color_output) draw_buffers[i] = GL_NONE;

237  }
