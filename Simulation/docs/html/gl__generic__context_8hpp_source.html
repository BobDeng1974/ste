61  void set_context_server_state(container<K, V> &m, const K &k, std::tuple<Ts...> &&v, FuncT *func, Args&&... args) const {
62  auto it = m.find(k);
63  auto args_tuple = std::tuple<std::remove_reference_t<Args>...>(args...);
64  bool execute = it == m.end() || !it->second.compare(v);
65 
66  if (logger) {
67  if (&m == &states) {
68  optional<context_state::state_type> old_state = none;
69  if (it != m.end() && execute && it->second.exists())
70  old_state = it->second.get_state();
71  logger->log_state_change(execute, std::make_pair(k, old_state));
72  }
73  else if (&m == &resources) logger->log_resource_change(execute, k);
74  else assert(false);
75  }
76 
77  if (!execute)
78  return;
79 
80  if (it == m.end())
81  it = m.emplace(std::make_pair(k, context_state())).first;

84  std::tuple<std::remove_reference_t<Args>...> params = t.get_weak<std::remove_reference_t<Args>...>();

167  void restore_states_from_logger(const gl_context_state_log &l) const {
168  for (auto &state : l.get_basic_states())
169  restore_default(state);
170  for (auto &state : l.get_states()) {
171  if (state.first.get_name() == StateName::SAMPLER_OBJECT)
172  bind_sampler(state.first.get_index0(), 0);
173  else if (state.first.get_name() == StateName::TRANSFORM_FEEDBACK_OBJECT)
174  bind_transform_feedback(state.first.get_index0(), 0);
175  else if (state.second) {
176  const context_state::state_type &s = state.second.get();
177  set_context_server_state(states, state.first, s);
178  }
179  else {
180  assert(false && "Can't restore.");
181  }
182  }
