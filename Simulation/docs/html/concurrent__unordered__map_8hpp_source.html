156  void copy_virtual_bucket(hash_table_guard_type &old_table_guard,
157  resize_data_guard_type &resize_guard,
158  int i, int chunks) {
159  unsigned chunk_size = resize_data::chunk_size;
160  unsigned j = static_cast<unsigned>(i) * chunk_size;
161  for (unsigned k = 0; k < chunk_size && j < old_table_guard->size; ++k, ++j) {
162  auto *virtual_bucket = &old_table_guard->buckets[j];
163  for (;;) {

262  template <typename ... Ts>
263  bool insert_update_into_virtual_bucket(concurrent_map_virtual_bucket &virtual_bucket,
264  hash_type hash,
265  const key_type &key,
266  float load_factor,
267  bool is_new_item_insert,
268  int depth,
269  Ts&&... val_args) {
270  bool request_resize = false;
271  for (int j = 0; j < N;) {
272  auto bucket_data = virtual_bucket.buckets[j].load();
273  if (!bucket_data) {
274  hash_type old_hash = virtual_bucket.hash[j].load(std::memory_order_relaxed);
275  if (virtual_bucket.hash[j].compare_exchange_strong(old_hash, hash, std::memory_order_acq_rel, std::memory_order_relaxed)) {
276  virtual_bucket.buckets[j].store(new concurrent_map_bucket_data(key, std::forward<Ts>(val_args)...), std::memory_order_release);
277  if (is_new_item_insert)
278  items.fetch_add(1, std::memory_order_relaxed);
279  return !request_resize;
280  }
281 
282  continue;
283  }
284 
285  hash_type old_hash = virtual_bucket.hash[j].load();
286  if (old_hash != hash || bucket_data->k != key) {
287  ++j; continue;
288  }
289  if (!is_new_item_insert)
290  return false;
291  bucket_data->v->emplace(std::memory_order_release, std::forward<Ts>(val_args)...);
292  return !request_resize;
293  }
294 
295  if (load_factor >= min_load_factor_for_resize && depth >= depth_threshold)
296  request_resize = true;
297 
298  auto next_ptr = virtual_bucket.next.load();
299  while (!next_ptr) {
300  auto new_next = new virtual_bucket_type;
301  if (virtual_bucket.next.compare_exchange_strong(next_ptr, new_next, std::memory_order_acq_rel, std::memory_order_acquire))
302  next_ptr = new_next;
303  assert(next_ptr);
304  }
305 
306  return !request_resize && insert_update_into_virtual_bucket(*next_ptr, hash, key, load_factor, is_new_item_insert, depth + 1, std::forward<Ts>(val_args)...);
307  }
