20 /*
21  * Defines the properties of an atmosphere
22  */
23 template <typename T>
24 struct atmospherics_properties {
25  using Vec = glm::tvec3<T>;
26 
27  // Center of the atmosphere and distance from center to where the atmosphere starts.
28  // Can be thought of as planet center and radius.
29  Vec center;
30  T radius;
31 
32  // Wave length dependent scattering coefficients for the Rayleigh scattering theory (m^-1)
33  // For Rayleigh scattering extinction = scattering, i.e. no absorption by atmosphere molecules is assumed. Absorption is handled for in the Mie aerosol scattering.
34  Vec rayleigh_scattering_coefficient;
35  // Scattering coefficient for the Mie scattering theory (m^-1)
36  T mie_scattering_coefficient;
37  // Absorption coefficient for the Mie scattering theory (m^-1)
38  T mie_absorption_coefficient;
39 
40  // Phase coefficient for the Mie scattering phase function
41  T phase;
42 
43  // Sea level athmospheric pressure, kPa
44  T ro0;
45  // Sea level temperature, Kelvin
46  T T0;
47  // Gravitation acceleration, m/s^2
48  T g;
49  // Ideal (universal) gas constant, J/(mol·K)
50  T R;
51  // Molar mass of the atmospheric air, kg/mol
52  T M;
53  // Molar mass of the aerosols in atmosphere, kg/mol
54  T M_aerosols;
55 
56 
57  /*
58  * Returns the height of a point
59  *
60  * @param P point in world coordinates
61  */
62  T height(const Vec &P) const {
63  return glm::length(P - center);
64  }
65 
66  /*
67  * Returns the scale height of the atmosphere air
68  */
69  T scale_height() const {
70  return R * T0 / (M * g);
71  }
72 
73  /*
74  * Returns the scale height of the aerosols
75  */
76  T scale_height_aerosols() const {
77  return R * T0 / (M_aerosols * g);
78  }
79 
80 
81  /*
82  * Returns the Mie extinction coefficient in m^-1
83  */
84  T mie_extinction_coeffcient() const {
85  return mie_scattering_coefficient + mie_absorption_coefficient;
86  }
87 
88  /*
89  * Returns the Rayleigh extinction coefficient in m^-1
90  */
91  Vec rayleigh_extinction_coeffcient() const {
92  return rayleigh_scattering_coefficient;
93  }
94 
95 
96  /*
97  * Returns the height at which atmospheric density becomes negligible
98  */
99  static T max_height(const T &H) {
100  constexpr T max_density = 1e-6;
101 
102  auto log_density = glm::log(max_density);
103  return -H * log_density;
104  }
105 
106  /*
107  * Returns the pressure at h using the exponential barometric law
108  *
109  * @param h Height in meters
110  */
111  T pressure(const T &h, const T &H) const {
112  return ro0 * glm::exp(-h / H);
113  }
114 };
