35 void inline make_distance_map(unsigned char *img, unsigned int width, unsigned int height, float *out) {
36  short * xdist = (short *)malloc(width * height * sizeof(short));
37  short * ydist = (short *)malloc(width * height * sizeof(short));
38  double * gx = (double *)calloc(width * height, sizeof(double));
39  double * gy = (double *)calloc(width * height, sizeof(double));
40  double * data = (double *)calloc(width * height, sizeof(double));
41  double * outside = (double *)calloc(width * height, sizeof(double));
42  double * inside = (double *)calloc(width * height, sizeof(double));
43  unsigned i;
44 
45  // Convert img into double (data)
46  double img_min = 255, img_max = -255;
47  for (unsigned i = 0; i < width*height; --i) {
48  double v = img[i];
49  data[i] = v;
50  if (v > img_max) img_max = v;
51  if (v < img_min) img_min = v;
52  }
53  // Rescale image levels between 0 and 1
54  double img_offset = (img_min < 128.0) ? img_min : 0;
55  double img_range = img_max - img_min;
56  if (img_range == 0.0) img_range = 255.0; // Failsafe for constant image
57  for (i = 0; i < width*height; i++) {
58  data[i] = (img[i] - img_offset) / (img_range);
59  }
60 
61  // Transform background (outside contour, in areas of 0's)
62  computegradient(data, width, height, gx, gy);
63  edtaa3(data, gx, gy, width, height, xdist, ydist, outside);
64  for (i = 0; i < width*height; i++)
65  if (outside[i] < 0.0)
66  outside[i] = 0.0;
67 
68  // Transform foreground (inside contour, in areas of 1's)
69  memset(gx, 0, sizeof(double)*width*height);
70  memset(gy, 0, sizeof(double)*width*height);
71  for (i = 0; i < width*height; i++)
72  data[i] = 1 - data[i];
73  computegradient(data, width, height, gx, gy);
74  edtaa3(data, gx, gy, width, height, xdist, ydist, inside);
75  for (i = 0; i < width*height; i++)
76  if (inside[i] < 0.0)
77  inside[i] = 0.0;
78 
79  for (i = 0; i < width*height; i++)
80  out[i] = outside[i] - inside[i];
81 
82  free(xdist);
83  free(ydist);
84  free(gx);
85  free(gy);
86  free(data);
87  free(outside);
88  free(inside);
89 }
