40  microfacet_transmission_fit(std::istream &is) {
41  fit_data = std::make_unique<transmission_fit_data>();
42 
43  is.read(reinterpret_cast<char*>(&fit_data->header), sizeof(fit_data->header) / sizeof(char));
44 
45  if (fit_data->header.version != Version)
46  throw microfacet_fit_error("Unsupported version");
47  if (fit_data->header.size == 0)
48  throw microfacet_fit_error("LUT size zero");
49 
50  std::size_t lut_row_size = fit_data->header.size;
51  fit_data->alloc_with_size(lut_row_size);
52 
53  auto lut_size = lut_row_size * lut_row_size * sizeof(DataType);
54  auto bytes_read = is.read(reinterpret_cast<char*>(fit_data->data.get()), lut_size / sizeof(char)).gcount();
55  if (bytes_read != lut_size)
56  throw microfacet_fit_error("Premature EOF");
57 
58  boost::crc_32_type crc_computer;
59  crc_computer.process_bytes(reinterpret_cast<const std::uint8_t*>(fit_data->data.get()), lut_size);
60  auto hash = crc_computer.checksum();
61 
62  if (fit_data->header.hash != hash)
63  throw microfacet_fit_error("Checksum mismatch");
64  }
