45  auto attrib_iterator_for_type(attrib_id_type id, const range_type &r) const {
46  for (auto it = attributes.begin(); it != attributes.end(); ++it) {
47  if (r.start + r.length < it->first.start)
48  break;
49  if (it->second->attrib_type() == id && it->first.overlaps(r))
50  return it;
51  }

53  return attributes.end();
54  }

56  attrib_storage_iterator remove_attrib(const attrib_storage_iterator &it) { return attributes.erase(it); }

58  void splice(attrib_storage_iterator it, const range_type &r) {
59  auto it_range = it->first;
60  auto it_attrib = std::move(it->second);
61  remove_attrib(it);
62 
63  auto length_before = static_cast<std::size_t>(std::max<int>(r.start - it_range.start, 0));
64  auto length_after = static_cast<std::size_t>(std::max<int>(it_range.start + it_range.length - r.length - r.start, 0));
65 
66  if (length_before)
67  attributes.insert(std::make_pair(range_type{ it_range.start, length_before }, std::unique_ptr<attrib_type>(it_attrib->clone())));
68  if (length_after)
69  attributes.insert(std::make_pair(range_type{ r.start + r.length, length_after }, std::unique_ptr<attrib_type>(it_attrib->clone())));
70  }

117  void add_attrib(const range_type &r, const attrib_type &a) {
118  for (;;) {
